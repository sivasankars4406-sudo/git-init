class Solution(object):
    def minimumOperations(self, grid):

        rows = len(grid)
        cols = len(grid[0])
        ans = 0
        for j in range(cols):

            for i in range(1, rows):
                if grid[i][j] <= grid[i-1][j]:
                    need = grid[i-1][j] + 1
                    ans += need - grid[i][j]
                    grid[i][j] = need
        return ans
